export default function App(props) {
  const [count, setCount] = React.useState(0)

  return (
    <div className='App'>
      <div>{count}</div>
      <button onClick={() => setCount(() => count + 1)}>add</button>
      <button onClick={() => setCount(count + 1)}>add</button>
      <button onClick={() => setCount(c => c + 1)}>add</button>
    </div>
  )
}

    # useEffect

    useEffect 在函数组件中执行副作用

    副作用是指一段和当前执行结果无关的代码，常用的副作用操作如数据获取、设置订阅、手动更改/获取组件中的 DOM元素。 useEffect 接受两个参数

    useEffect(callback, dependencies)
    1

    第一个参数是要执行的函数 callback，第二个参数是可选的依赖数组 dependencies。 示例

    import React from 'react'

export default function App() {
  const [count, setCount] = React.useState(0)

  React.useEffect(()=>{
    console.log(count)
  })

  return <div onClick={() => setCount(count+1)}>{count}</div>
}

    每当 count 发生变化后，useEffect 副作用函数就会输出 count，由于没传入依赖数组，则每次 render 后执行

    第二个参数给空数组的话，只会在第一次加载组件时执行，常用于首次数据请求。

    import React from 'react'

export default function App() {
    const [data, setData] = React.useState('')
    async function fetchData() {
        const data = await fetch('xxx').text()
        setData(data)
    }

    React.useEffect(() => {

        fetchData()
    }, [])

    return  <div>{data}</div>
}

    # useEffect 总结

    • 每次 render 后执行：不提供第二个依赖项参数。比如 useEffect(() => {})
    • 仅初始化 render 后执行：提供一个空数组作为依赖项。比如 useEffect(() => {}, [])
    • 第一次以及依赖项发生变化后执行：提供依赖项数组。比如 useEffect(() => {}, [deps])
    • 组件卸载执行：返回一个回调函数。比如 useEffect(() => { return () => {} }, [])

    # useMemo

    useMemo 定义的创建函数只会在依赖项改变时才重新计算，有助于每次渲染时不会重复的高开销的计算，而接收这个计算值作为属性的组件，也不会频繁地需要重新渲染。类似与 Vue 中的 computed

    示例：

    const memoizedValue = useMemo(() => computeExpensiveValue(a, b), [a, b])
    1

    useMemo 本质上就像一个缓存，而依赖项是缓存失效策略。

    不仅能对数据进行缓存，对于纯组件也是能够缓存的。使用 memo 对组件进行包裹即可，例如 export default React.memo(Children)

    # useCallback

    useCallback 定义的回调只会在依赖数组项改变时重新执行这个回调，保证了组件不会创建重复的回调函数。而接收这个回调函数作为属性的组件，也不会频繁地需要重新渲染。

    useMemo和useCallback的作用是一样的，前者缓存值后者缓存函数

    const callbackFun = useCallback(()=>setCount((cur=>cur+1)), [])
    1

    # useRef

    useRef 返回一个ref对象，这个对象在组件生命周期内一直存在。

    • 保存DOM，可以获取DOM节点信息
    • 共享数据 保存DOM信息：
    const InputFocusButton = () =>{
  const inputRef = React.useRef(null);
  const handelClick = () =>{
    inputRef.current.focus()
  }
  return(
    <>
    <input ref={inputRef}/>
    <button onClick={handelClick}>点击获取焦点</button>
    </>
  )
}

    # useReducer

    const [state, dispatch] = useReducer(reducer, initialArg, init)

    第一个参数 reducer 是函数 (state, action) => newState，接受当前的 state 和操作行为。第二个参数 initialArg 是状态初始值。第三个参数 init 是初始化函数。

    import React form 'react';
const initValue = {count : 0};
const reduce = (state,action) =>{
  switch(action.type){
    case 'increment':
    return {count:state.count+1}
    case 'decrement':
    return {count:state.count-1}
    default:
    break
  }
}

const Count = () =>{
  const [state,dispatch] = useReducer(reduce, initValue)
  return(
    <>
    {state.count}
    <button onClick={dispatch({type:'increment'})}>加</button>
    <button onClick={dispatch({type:'decrement'})}>减</button>
    </>
  )
}

    # 自定义hooks

    • 命名以use开头
    • 函数内部调用其他的Hook。 例如：
    import { useEffect, useState } from "react"

const useLocalStorage = (key, defaultValue) => {
    const [value, setValue] = useState(defaultValue)
    useEffect(() => {
        window.localStorage.setItem(key, value)
    },[key, value])
    return [value, setValue]
}

function App() {
  const [value, setValue] = useLocalStorage('key', 'react')

  return (
    <div>

    <button onClick={() => {
        //点击修改value，会自动保存localstorage
        setValue('vue')
      }}>点击</button>
      <div>{ value }</div>
    </div>
  )
}
export default App
